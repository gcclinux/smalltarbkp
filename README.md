# smalltarbkp
version: 5.0 - Build 28

INSTALL:
$ cd  ${HOME}<br>
$ curl -L https://bit.ly/3qctuvQ -o smalltarbkp<br>
$ chmod +x smalltarbkp<br>
$ alias smalltarbkp='${HOME}/smalltarbkp'<br>
$ ource ~/.bashrc<br>
$ smalltarbkp --setup<br>
